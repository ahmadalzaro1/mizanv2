**Create `training-api.ts`:**
```typescript
import { api } from './api'
import type { TrainingSession, SessionItem } from './types'

export async function createSession(): Promise<TrainingSession> {
  const res = await api.post<TrainingSession>('/api/training/sessions')
  return res.data
}

export async function getSession(sessionId: string): Promise<TrainingSession> {
  const res = await api.get<TrainingSession>(`/api/training/sessions/${sessionId}`)
  return res.data
}

export async function submitLabel(
  sessionId: string,
  itemId: string,
  label: { moderator_label: 'hate' | 'not_hate'; moderator_hate_type?: string }
): Promise<SessionItem> {
  const res = await api.put<SessionItem>(
    `/api/training/sessions/${sessionId}/items/${itemId}`,
    label
  )
  return res.data
}

export async function listSessions(): Promise<TrainingSession[]> {
  const res = await api.get<{ sessions: TrainingSession[] }>('/api/training/sessions')
  return res.data.sessions
}
```

### Task 2: Create TweetCard and ProgressBar components
**Files**: `mizan/frontend/src/components/TweetCard.tsx`, `mizan/frontend/src/components/ProgressBar.tsx`

**TweetCard** — displays a single Arabic tweet in a card designed for reading:
- Large Arabic text (`text-xl` or `text-2xl`) in Tajawal font
- Source dataset badge (small, muted — e.g., "JHSC" / "Let-Mi")
- Dialect badge if available
- White card with subtle shadow on surface background
- Adequate padding for comfortable reading

```tsx
interface TweetCardProps {
  text: string
  sourceDataset: string
  dialect: string | null
}
```

**ProgressBar** — shows training progress:
- Visual bar: filled portion = labeled items / total items
- Text: "٧ من ٢٠" (7 of 20) — use Arabic position text
- Current position indicator
- Color: navy fill on light gray background
- Progress bar fills left-to-right (standard visual direction even in RTL interfaces)

```tsx
interface ProgressBarProps {
  current: number  // current position (1-based)
  total: number
  labeled: number  // how many have been labeled
}
```

### Task 3: Create LabelSelector component
**File**: `mizan/frontend/src/components/LabelSelector.tsx`

Two-step label selection:

**Step 1**: Two large buttons side by side:
- "خطاب كراهية" (Hate) — red-tinted background (`bg-red-50 border-red-300 hover:bg-red-100`)
- "ليس كراهية" (Not hate) — green-tinted background (`bg-green-50 border-green-300 hover:bg-green-100`)
- Selected state: filled with stronger color

**Step 2** (shown only when "hate" selected): A grid of 9 category buttons:
- Each shows Arabic label from `HATE_CATEGORIES`
- Selected category is visually highlighted (navy border + background tint)
- Grid: 3 columns × 3 rows on desktop

**Submit button**: Appears when selection is complete (`not_hate` selected, OR `hate` + category selected). Label: "إرسال" (Submit). Navy background, white text, full-width.

```tsx
interface LabelSelectorProps {
  onSubmit: (label: 'hate' | 'not_hate', hateType?: string) => void
  disabled?: boolean
  initialLabel?: 'hate' | 'not_hate' | null
  initialHateType?: string | null
}
```

Component manages its own local state for the two-step flow. Accepts `initialLabel`/`initialHateType` to restore state when navigating back.
